public class Main {

    public static void main(String[] args) {


        /**
         * 匿名内部类
         * 1、匿名内部类没有类型名称、实例对象名称
         * 2、编译后的文件命名:外部类$数字.class
         * 3、无法使用private、public.protected、abstract、static修饰4、无法编写构造方法，可以添加构造代码块
         * 5、不能出现静态成员
         * 6、匿内部类可以实现接口也可以继承父类，但是不可兼得
         */
        System.out.println(new Person(){
            @Override
            public String getString(){
                return "456";
            }
        }.getString());
    }
    
}
